FIND_PACKAGE(PkgConfig REQUIRED)

PKG_CHECK_MODULES(SKIA REQUIRED skia)
INCLUDE_DIRECTORIES(${SKIA_INCLUDE_DIRS})
LINK_DIRECTORIES(${SKIA_LIBRARY_DIRS})

PKG_CHECK_MODULES(FFTW3 REQUIRED fftw3)
INCLUDE_DIRECTORIES(${FFTW3_INCLUDE_DIRS})
LINK_DIRECTORIES(${FFTW3_LIBRARY_DIRS})

PKG_CHECK_MODULES(LIBPULSESIMPLE REQUIRED libpulse-simple)
INCLUDE_DIRECTORIES(${LIBPULSESIMPLE_INCLUDE_DIRS})
LINK_DIRECTORIES(${LIBPULSESIMPLE_LIBRARY_DIRS})

set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS}")
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} ${CMAKE_C_FLAGS}")
set(CMAKE_LINK_FLAGS "${CMAKE_LINK_FLAGS}")

include_directories(
	${CMAKE_CURRENT_BINARY_DIR}
	${CMAKE_SOURCE_DIR}/show
	${CMAKE_SOURCE_DIR}/tale
	${CMAKE_SOURCE_DIR}/tool
	${CMAKE_BINARY_DIR}/tool
	${CMAKE_SOURCE_DIR}/mote
	${CMAKE_SOURCE_DIR}/mosi
	${CMAKE_SOURCE_DIR}/apps
	${CMAKE_SOURCE_DIR}/thirds
	${CMAKE_SOURCE_DIR}/thirds/poly2tri
	)

add_definitions(-DNEMOUX_MIROBACK_RESOURCES="${CMAKE_INSTALL_PREFIX}/share/nemobacks/miroback")

add_executable(nemoback-miro
	miroback.c
	mirotap.c
	)

TARGET_LINK_LIBRARIES(nemoback-miro
	pthread
	nemoapps
	${NEMOUX_COMMON_LIBRARIES}
	${SKIA_LIBRARIES}
	${FFTW3_LIBRARIES}
	${LIBPULSESIMPLE_LIBRARIES}
	)

set(RESOURCE_FILES
	${CMAKE_CURRENT_SOURCE_DIR}/resources/touch-1.svg
	${CMAKE_CURRENT_SOURCE_DIR}/resources/touch-2.svg
	${CMAKE_CURRENT_SOURCE_DIR}/resources/touch-3-1.svg
	${CMAKE_CURRENT_SOURCE_DIR}/resources/touch-3-2.svg
	${CMAKE_CURRENT_SOURCE_DIR}/resources/touch-3-3.svg
	${CMAKE_CURRENT_SOURCE_DIR}/resources/touch-3-4.svg
	)

INSTALL(TARGETS nemoback-miro DESTINATION bin)
INSTALL(FILES ${RESOURCE_FILES} DESTINATION share/nemobacks/miroback)
